```python
def build_default_lob_feature_set_v3() -> FeatureSet:
    """LOB feature set v3: v2 + multi-window EMA aggregation features.

    Indices 0-21: identical to v2 (backward compatible).

    New features (Phase 1 INF-3, HAR-style aggregation):
    - ofi_l1_ema5s [22]: EMA(ofi_l1_raw, span=40 ticks / ~5s)
    - ofi_l1_ema30s [23]: EMA(ofi_l1_raw, span=240 ticks / ~30s)
    - imbalance_ema5s_ppm [24]: EMA(l1_imbalance_ppm, span=40 ticks / ~5s)
    - spread_ema30s [25]: EMA(spread_scaled, span=240 ticks / ~30s)
    - spread_ema300s [26]: EMA(spread_scaled, span=2400 ticks / ~300s)
    """
    v2 = build_default_lob_feature_set_v2()
    return FeatureSet(
        feature_set_id=FEATURE_SET_VERSION,
        schema_version=3,
        features=v2.features
        + (
            FeatureSpec("ofi_l1_ema5s", "i64", source_kind="book", warmup_min_events=40),
            FeatureSpec("ofi_l1_ema30s", "i64", source_kind="book", warmup_min_events=240),
            FeatureSpec("imbalance_ema5s_ppm", "i64", scale=1_000_000, source_kind="book", warmup_min_events=40),
            FeatureSpec("spread_ema30s", "i64", scale=10_000, source_kind="book", warmup_min_events=240),
            FeatureSpec("spread_ema300s", "i64", scale=10_000, source_kind="book", warmup_min_events=2400),
        ),
    )
```

```python
"""Tests for FeatureEngine v3 multi-window EMA aggregation features."""
from __future__ import annotations

import pytest
from hft_platform.feature.engine import FeatureEngine


def _make_stats(symbol: str, best_bid: int, best_ask: int, bid_depth: int, ask_depth: int, ts: int = 1_000_000_000):
    """Minimal LOBStatsEvent-compatible tuple (see _StatsTupleProxy layout)."""
    mid_x2 = best_bid + best_ask
    spread = best_ask - best_bid
    imb = (bid_depth - ask_depth) / max(bid_depth + ask_depth, 1)
    return (symbol, ts, mid_x2, spread, imb, best_bid, best_ask, bid_depth, ask_depth)


def _make_event(bids, asks):
    """Minimal BidAskEvent-like object with bids/asks arrays."""
    import numpy as np
    class _E:
        pass
    e = _E()
    e.bids = np.array(bids, dtype=np.int64) if bids else np.zeros((0, 2), dtype=np.int64)
    e.asks = np.array(asks, dtype=np.int64) if asks else np.zeros((0, 2), dtype=np.int64)
    return e


class TestV3EmaFeatures:
    """Verify 5 new EMA aggregation features [22]-[26]."""

    def _make_engine(self) -> FeatureEngine:
        return FeatureEngine(feature_set_id="lob_shared_v3", kernel_backend="python")

    def test_v3_produces_27_features(self) -> None:
        engine = self._make_engine()
        stats = _make_stats("SYM", 100_0000, 101_0000, 50, 50)
        event = _make_event([[100_0000, 50]], [[101_0000, 50]])
        engine.process_lob_update(event, stats)
        vals = engine.get_feature_tuple("SYM")
        assert vals is not None
        assert len(vals) == 27

    def test_ema_converges_to_constant_input(self) -> None:
        """Feed constant OFI=100, imbalance=500000, spread=30000 for 500 ticks.
        All EMAs should converge close to the constant value."""
        engine = self._make_engine()
        for i in range(500):
            stats = _make_stats("SYM", 100_0000, 103_0000, 50, 50, ts=1_000_000_000 + i * 125_000_000)
            event = _make_event(
                [[100_0000, 50], [99_0000, 30], [98_0000, 20], [97_0000, 10], [96_0000, 5]],
                [[103_0000, 50], [104_0000, 30], [105_0000, 20], [106_0000, 10], [107_0000, 5]],
            )
            engine.process_lob_update(event, stats)

        vals = engine.get_feature_tuple("SYM")
        assert vals is not None
        # spread_scaled = 3_0000 (ask - bid)
        spread_ema30s = vals[25]  # should converge to ~30000
        spread_ema300s = vals[26]  # should converge to ~30000 (but slower)
        assert abs(spread_ema30s - 3_0000) < 500, f"spread_ema30s={spread_ema30s} not near 30000"
        # spread_ema300s needs 2400 ticks to fully converge; with 500 ticks it's partial
        # Just verify it's positive and trending toward 30000
        assert spread_ema300s > 0

    def test_ema_alpha_constants(self) -> None:
        """Verify alpha = 2/(N+1) for each window."""
        engine = self._make_engine()
        assert abs(engine._alpha_5s - 2.0 / 41.0) < 1e-10
        assert abs(engine._alpha_30s - 2.0 / 241.0) < 1e-10
        assert abs(engine._alpha_300s - 2.0 / 2401.0) < 1e-10

    def test_warmup_mask_respects_ema_windows(self) -> None:
        """Features [22]-[24] (5s) should be warm after 40 ticks.
        Features [25] (30s) after 240. Features [26] (300s) after 2400."""
        engine = self._make_engine()
        for i in range(45):
            stats = _make_stats("SYM", 100_0000, 101_0000, 50, 50, ts=1_000_000_000 + i * 125_000_000)
            event = _make_event([[100_0000, 50]], [[101_0000, 50]])
            fe = engine.process_lob_update(event, stats)

        # After 45 ticks: 5s features [22]-[24] should be warm (warmup=40)
        assert fe is not None
        mask = fe.warmup_ready_mask
        assert mask & (1 << 22), "ofi_l1_ema5s should be warm at tick 45"
        assert mask & (1 << 24), "imbalance_ema5s_ppm should be warm at tick 45"
        # 30s features [23],[25] should NOT be warm yet (warmup=240)
        assert not (mask & (1 << 23)), "ofi_l1_ema30s should NOT be warm at tick 45"
        assert not (mask & (1 << 25)), "spread_ema30s should NOT be warm at tick 45"

    def test_v2_features_unchanged(self) -> None:
        """First 22 features must be identical between v2 and v3 engines."""
        engine_v2 = FeatureEngine(feature_set_id="lob_shared_v2", kernel_backend="python")
        engine_v3 = self._make_engine()
        for i in range(10):
            stats = _make_stats("SYM", 100_0000, 101_0000, 50, 50, ts=1_000_000_000 + i * 125_000_000)
            event = _make_event(
                [[100_0000, 50], [99_0000, 30], [98_0000, 20], [97_0000, 10], [96_0000, 5]],
                [[101_0000, 50], [102_0000, 30], [103_0000, 20], [106_0000, 10], [107_0000, 5]],
            )
            engine_v2.process_lob_update(event, stats)
            engine_v3.process_lob_update(event, stats)

        v2_vals = engine_v2.get_feature_tuple("SYM")
        v3_vals = engine_v3.get_feature_tuple("SYM")
        assert v2_vals is not None and v3_vals is not None
        for i in range(22):
            assert v2_vals[i] == v3_vals[i], f"Feature [{i}] diverged: v2={v2_vals[i]} v3={v3_vals[i]}"

    def test_ofi_ema_responds_to_flow_direction(self) -> None:
        """When OFI is consistently positive, ofi_l1_ema5s should be positive."""
        engine = self._make_engine()
        # Drive consistent positive OFI by increasing bid depth
        for i in range(100):
            bid_qty = 50 + i  # growing bid depth → positive OFI
            stats = _make_stats("SYM", 100_0000, 101_0000, bid_qty, 50, ts=1_000_000_000 + i * 125_000_000)
            event = _make_event([[100_0000, bid_qty]], [[101_0000, 50]])
            engine.process_lob_update(event, stats)

        vals = engine.get_feature_tuple("SYM")
        assert vals is not None
        ofi_ema5s = vals[22]
        # With growing bid depth, OFI should be positive, so EMA should be positive
        assert ofi_ema5s > 0, f"ofi_l1_ema5s={ofi_ema5s} should be positive with growing bid depth"
```
